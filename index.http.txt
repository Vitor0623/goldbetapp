<script>
    // Lógica de Inicialização
    document.addEventListener("DOMContentLoaded", () => {
        const savedUser = localStorage.getItem("goldbet_user");
        if (savedUser) {
            mostrarJogo(JSON.parse(savedUser));
        }
    });

    function cadastrar() {
        const nome = document.getElementById('regNome').value;
        const email = document.getElementById('regEmail').value;
        
        if(!nome || !email) return alert("Preencha os campos!");

        const novoUsuario = {
            nome: nome.split(' ')[0], // Pega só o primeiro nome
            email: email,
            saldo: 100.00 // Bônus inicial
        };

        localStorage.setItem("goldbet_user", JSON.stringify(novoUsuario));
        mostrarJogo(novoUsuario);
    }

    function mostrarJogo(user) {
        document.getElementById('authScreen').classList.add('hidden');
        document.getElementById('gameScreen').classList.remove('hidden');
        document.getElementById('userNome').innerText = user.nome;
        document.getElementById('saldo').innerText = user.saldo.toFixed(2);
    }

    function apostar() {
        let user = JSON.parse(localStorage.getItem("goldbet_user"));
        const valor = parseFloat(document.getElementById('valorAposta').value);
        const resDiv = document.getElementById('resultado');

        if (valor > user.saldo) {
            resDiv.innerHTML = "<span class='lose'>Saldo Insuficiente!</span>";
            return;
        }

        // RNG - Sorteio (40% vitória)
        if (Math.random() > 0.60) {
            user.saldo += valor;
            resDiv.innerHTML = <span class='win'>GANHOU! + R$ ${valor.toFixed(2)}</span>;
        } else {
            user.saldo -= valor;
            resDiv.innerHTML = <span class='lose'>PERDEU! Tente de novo.</span>;
        }

        document.getElementById('saldo').innerText = user.saldo.toFixed(2);
        localStorage.setItem("goldbet_user", JSON.stringify(user));
    }

    function logout() {
        localStorage.removeItem("goldbet_user");
        location.reload();
    }
</script>
