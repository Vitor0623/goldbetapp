<script>
    let state = {
        logged: false,
        name: "",
        balance: 100.00,
        isPlaying: false
    };

    function handleAuth() {
        const user = document.getElementById('username').value;
        if(user.length < 3) return alert("Digite um nome válido!");
        
        state.logged = true;
        state.name = user;
        
        document.getElementById('loginScreen').classList.add('hidden');
        document.getElementById('mainHeader').classList.remove('hidden');
        document.getElementById('gameScreen').classList.remove('hidden');
        document.getElementById('navBar').classList.remove('hidden');
        
        updateUI();
    }

    function updateUI() {
        document.getElementById('userBalance').innerText = state.balance.toFixed(2);
    }

    function playGame() {
        if(state.isPlaying) return;
        
        const amount = parseFloat(document.getElementById('betAmount').value);
        if(amount > state.balance) return alert("Saldo insuficiente!");
        if(amount <= 0) return alert("Valor inválido!");

        state.isPlaying = true;
        state.balance -= amount;
        updateUI();

        const display = document.getElementById('displayStatus');
        const btn = document.getElementById('btnPlay');
        
        btn.innerText = "SORTEANDO...";
        btn.disabled = true;
        display.style.color = "white";

        let counter = 1.00;
        let timer = setInterval(() => {
            counter += 0.1;
            display.innerText = counter.toFixed(2) + "x";
            if(counter > 2.0) clearInterval(timer);
        }, 50);

        setTimeout(() => {
            clearInterval(timer);
            const win = Math.random() > 0.6; // 40% chance
            
            if(win) {
                const prize = amount * 2;
                state.balance += prize;
                display.innerText = "GANHOU!";
                display.style.color = "#27ae60";
                addHistory("Vitoria", amount * 2);
            } else {
                display.innerText = "PERDEU";
                display.style.color = "#e74c3c";
                addHistory("Derrota", amount);
            }

            state.isPlaying = false;
            btn.innerText = "COMEÇAR RODADA";
            btn.disabled = false;
            updateUI();
        }, 1500);
    }

    function addHistory(type, val) {
        const list = document.getElementById('historyList');
        const item = document.createElement('span');
        item.style.marginRight = "10px";
        item.style.fontSize = "12px";
        item.style.padding = "2px 8px";
        item.style.borderRadius = "5px";
        item.style.background = type === "Vitoria" ? "#27ae6033" : "#e74c3c33";
        item.innerText = (type === "Vitoria" ? "✔ " : "✘ ") + "R$ " + val.toFixed(2);
        list.prepend(item);
    }

    function addFunds() {
        alert("Integração de PIX: Para ativar depósitos reais, você precisa de uma chave de API (SuitPay ou Mercado Pago). Por enquanto, ganhando +R$ 50 de teste!");
        state.balance += 50;
        updateUI();
    }
</script>
